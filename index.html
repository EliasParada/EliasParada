function rotatePart(degrees, part, joint) {
    var jointRect = joint.getBoundingClientRect();
    var jointCenterX = jointRect.left + jointRect.width / 2;
    var jointCenterY = jointRect.top + jointRect.height / 2;
    
    part.style.transformOrigin = jointCenterX + "px " + jointCenterY + "px";
    part.style.transform = "rotate(" + degrees + "deg)";
}
